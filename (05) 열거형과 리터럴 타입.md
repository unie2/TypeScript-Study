## 1. 열거형 (Enum)
  - 연관된 아이템들을 함께 묶어서 표현할 수 있는 수단이다.

```typescript
// Enum 정의
enum GenderType { // 숫자형 열거형 사용 시
  Male, // 0의 값을 가진다.
  Female // 1의 값을 가진다.
}
enum GenderType { // 문자열 열거형 사용 시
  Male = 'male',
  Female = 'female',
  genderNeutral = 'genderNeutral'
}

let student1 = {
  studentID: 121212,
  studentName: 'Janet Jakcson',
  age: 30,
  gender: GenderType.Female,
  subject: 'Mongo DB',
  courseCompleted: false
}

// Student 이름의 인터페이스 정의
interface Student { 
  readonly studentID: number,
  studentName: string,
  age?: number,
  gender: GenderType,
  subject: string,
  courseCompleted: boolean,
  //addComment (comment: string): string;
  addComment?: (comment: string) => string
}

// 타입 명시를 적용한 함수 작성
function getStudentDetails(studentID: number): Student { 
  // 반환타입을 Student 인터페이스로 지정
  return {
    studentID: 123456,
    studentName: 'Mark Jacobs',
    // gender: 'male', // 에러 발생 (GenderType이 아닌 string 값이기 때문에)
    gender: GenderType.Male,
    subject: 'Node JS',
    courseCompleted: true
  };
} 

function saveStudentDetails (student: Student): void {

}

saveStudentDetails(student1);

```

## 2. 리터럴 타입 (Literal Type)
  - 메소드는 객체 내에서 선언된 함수이다.
```typescript

let student1 = {
  studentID: 121212,
  studentName: 'Janet Jakcson',
  age: 30,
  gender: 'female',
  subject: 'Mongo DB',
  courseCompleted: false
}

// Student 이름의 인터페이스 정의
interface Student { 
  readonly studentID: number,
  studentName: string,
  age?: number,
  gender: 'male' | 'female' | 'genderNeutral', // 리터럴 타입 사용
  subject: string,
  courseCompleted: boolean,
  //addComment (comment: string): string;
  addComment?: (comment: string) => string
}

// 타입 명시를 적용한 함수 작성
function getStudentDetails(studentID: number): Student { 
  // 반환타입을 Student 인터페이스로 지정
  return {
    studentID: 123456,
    studentName: 'Mark Jacobs',
    gender: 'male', // 리터럴 타입에 정의된 string 값으로 설정
    subject: 'Node JS',
    courseCompleted: true
  };
} 

function saveStudentDetails (student: Student): void {

}

saveStudentDetails(student1);
```
